function eliminarCliente() { // Obtener DNI a eliminar const dniEliminar = $("#eliminarForm input[name=dniEliminar]").val(); // Llamada a la API para eliminar el cliente $.ajax({ type: "DELETE", url: `/clientes/${dniEliminar}`, success: function (data) { // Actualizar la lista después de eliminar el cliente cargarListaClientes(); console.log(data.mensaje); }, error: function (error) { console.error("Error al eliminar el cliente:", error); } }); // Evento para manejar la eliminación de un cliente $("#eliminarForm").submit(function (event) { event.preventDefault(); eliminarCliente(); });
